// Test 1 : IF dans WHILE
PROGRAM Test
DECL
    INTEGER : A, B, C ;
ENDDECL
BEGIN
    A = 5 ;
    B = 0 ;
    WHILE (A > 0) {
        IF (A > 2) {
            C = A + B ;
        }
        B = B + 1 ;
        A = A - 1 ;
    }
END

//Test 2 : IF/ELSE dans WHILE
PROGRAM Test
DECL
    INTEGER : A, B, C ;
ENDDECL
BEGIN
    A = 5 ;
    B = 0 ;
    WHILE (A > 0) {
        IF (A > 2) {
            C = A + B ;
        } ELSE {
            C = 0 ;
        }
        A = A - 1 ;
    }
END

//Test 3 : FOR dans FOR
PROGRAM Test
DECL
    INTEGER : A, B, C ;
ENDDECL
BEGIN
    C = 0 ;
    FOR (A : 1 : 5 : 1) {
        FOR (B : 1 : 3 : 1) {
            C = C + 1 ;
        }
    }
END

//Test 4 : IF dans IF (le plus critique)
PROGRAM Test
DECL
    INTEGER : A, B, C ;
ENDDECL
BEGIN
    A = 5 ;
    B = 3 ;
    C = 0 ;
    IF (A > 0) {
        IF (B > 0) {
            C = A + B ;
        }
        A = A - 1 ;
    }
END
